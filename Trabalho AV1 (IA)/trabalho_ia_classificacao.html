import numpy as np
import matplotlib.pyplot as plt

# Carregar os dados
dados = np.loadtxt("TrabalhoAV1/dados/EMGsDataset.csv", delimiter=",")

# Separar caractersticas e rtulos
X = dados[:2, :].T  # Transposta para ter formato (50000, 2)
y = dados[2, :].astype(int)  # Converter para inteiro

# 1. Pr-processamento dos dados
# Normalizar os dados (importante para KNN)
X_normalized = (X - X.min(axis=0)) / (X.max(axis=0) - X.min(axis=0))

# 2. Dividir em conjuntos de treino e teste (80% treino, 20% teste)
np.random.seed(42)
indices = np.random.permutation(len(X))
train_size = int(0.8 * len(X))

X_train = X_normalized[indices[:train_size]]
y_train = y[indices[:train_size]]
X_test = X_normalized[indices[train_size:]]
y_test = y[indices[train_size:]]

# 3. Implementao do KNN
def knn_predict(X_train, y_train, X_test, k=5):
    y_pred = np.zeros(X_test.shape[0], dtype=int)
    
    for i, test_sample in enumerate(X_test):
        # Calcular distncias euclidianas
        distances = np.sqrt(np.sum((X_train - test_sample)**2, axis=1))
        
        # Obter os k vizinhos mais prximos
        k_indices = np.argpartition(distances, k)[:k]
        k_nearest_labels = y_train[k_indices]
        
        # Votao para determinar a classe
        y_pred[i] = np.bincount(k_nearest_labels).argmax()
    
    return y_pred

# 4. Fazer previses e avaliar o modelo
k = 5  # Nmero de vizinhos
y_pred = knn_predict(X_train, y_train, X_test, k)

# Calcular acurcia
accuracy = np.mean(y_pred == y_test)
print(f"Acurcia do KNN (k={k}): {accuracy:.2%}")

# 5. Visualizao dos resultados
# Plotar os dados de treino
plt.figure(figsize=(12, 6))
plt.subplot(1, 2, 1)
for class_id in np.unique(y_train):
    plt.scatter(X_train[y_train == class_id, 0], 
                X_train[y_train == class_id, 1], 
                label=f'Classe {class_id}', alpha=0.6)
plt.title('Dados de Treino (Normalizados)')
plt.xlabel('Sensor 1 - Corrugador do Superclio')
plt.ylabel('Sensor 2 - Zigomtico Maior')
plt.legend()

# Plotar os dados de teste com as previses
plt.subplot(1, 2, 2)
for class_id in np.unique(y_test):
    plt.scatter(X_test[y_test == class_id, 0], 
                X_test[y_test == class_id, 1], 
                label=f'Classe {class_id}', alpha=0.6)
plt.title('Dados de Teste com Previses')
plt.xlabel('Sensor 1 - Corrugador do Superclio')
plt.legend()

plt.tight_layout()
plt.show()

# Matriz de confuso
def plot_confusion_matrix(y_true, y_pred, classes):
    cm = np.zeros((len(classes), len(classes)), dtype=int)
    
    for i in range(len(y_true)):
        cm[y_true[i]-1, y_pred[i]-1] += 1
    
    plt.figure(figsize=(8, 6))
    plt.imshow(cm, interpolation='nearest', cmap=plt.cm.Blues)
    plt.title('Matriz de Confuso')
    plt.colorbar()
    
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes)
    plt.yticks(tick_marks, classes)
    
    for i in range(len(classes)):
        for j in range(len(classes)):
            plt.text(j, i, str(cm[i, j]),
                     horizontalalignment="center",
                     color="white" if cm[i, j] > cm.max()/2 else "black")
    
    plt.ylabel('Rtulo Verdadeiro')
    plt.xlabel('Rtulo Predito')
    plt.tight_layout()
    plt.show()

plot_confusion_matrix(y_test, y_pred, ['1-Neutro', '2-Sorriso', '3-Sobrancelhas', '4-Surpreso', '5-Rabugento'])

Quais suas recomendaes quanto aos modelos de gauss bayas

Para o problema de classificao de expresses faciais usando os dados de EMG, modelos Gaussianos Bayesianos podem ser uma boa escolha, especialmente devido natureza contnua dos sinais dos sensores e possvel distribuio normal dos dados. Aqui esto minhas recomendaes:

1. Anlise de Adequao dos Modelos Gaussianos Bayesianos

  • Prs:

    • Eficincia computacional: Rpidos para treinar em datasets mdios (50k amostras).

    • Interpretabilidade: Probabilidades claras por classe.

    • Bom desempenho se os dados seguirem distribuio normal ou forem transformveis (ex.: log-transform).

  • Contras:

    • Sensibilidade a outliers: Requer normalizao robusta.

    • Pressuposto de independncia: Assume que as features (sensores) so independentes, o que pode no ser verdade (msculos faciais podem ativar juntos).

class GaussianNaiveBayes:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.mean = {}
        self.var = {}
        self.prior = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.mean[c] = np.mean(X_c, axis=0)
            self.var[c] = np.var(X_c, axis=0)
            self.prior[c] = X_c.shape[0] / X.shape[0]
    
    def predict(self, X):
        posteriors = []
        
        for c in self.classes:
            prior = np.log(self.prior[c])
            likelihood = -0.5 * np.sum(np.log(2 * np.pi * self.var[c]) + 
                                     ((X - self.mean[c]) ** 2) / self.var[c], axis=1)
            posterior = prior + likelihood
            posteriors.append(posterior)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

B. Anlise de Discriminante Gaussiano (GDA)

  • Vantagem: Captura correlaes entre os sensores.

class GaussianDiscriminantAnalysis:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.mean = {}
        self.shared_cov = np.cov(X.T)  # Covarincia compartilhada
        self.prior = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.mean[c] = np.mean(X_c, axis=0)
            self.prior[c] = X_c.shape[0] / X.shape[0]
    
    def predict(self, X):
        posteriors = []
        inv_cov = np.linalg.inv(self.shared_cov)
        
        for c in self.classes:
            diff = X - self.mean[c]
            likelihood = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            prior = np.log(self.prior[c])
            posteriors.append(prior + likelihood)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

class MQOMultinomial:
    def fit(self, X, Y):
        self.W = np.linalg.inv(X.T @ X) @ X.T @ Y  # Soluo fechada: W = (X^T X)^-1 X^T Y
    
    def predict(self, X):
        scores = X @ self.W
        return np.argmax(scores, axis=1) + 1  # +1 para classes 1-5

Assume covarincias diferentes por classe.

python
Copy
class GDA:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.covs = {}
        self.priors = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.covs[c] = np.cov(X_c.T)  # Covarincia por classe
            self.priors[c] = len(X_c) / len(X)
    
    def predict(self, X):
        posteriors = []
        for c in self.classes:
            diff = X - self.means[c]
            inv_cov = np.linalg.inv(self.covs[c])
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            norm = 1 / np.sqrt((2 * np.pi)**2 * np.linalg.det(self.covs[c]))
            likelihood = norm * np.exp(exponent)
            posteriors.append(likelihood * self.priors[c])
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

3. Classificador Gaussiano com Covarincias Iguais (LDA)

Assume mesma covarincia para todas as classes.

python
Copy
class LDA:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.priors = {}
        self.shared_cov = np.zeros((X.shape[1], X.shape[1]))
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
            self.shared_cov += (len(X_c) - 1) * np.cov(X_c.T)
        
        self.shared_cov /= (len(X) - len(self.classes))  # Mdia ponderada
    
    def predict(self, X):
        posteriors = []
        inv_cov = np.linalg.inv(self.shared_cov)
        for c in self.classes:
            diff = X - self.means[c]
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            prior = np.log(self.priors[c])
            posteriors.append(prior + exponent)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

5. Classificador Gaussiano Regularizado (Friedman)

Combina covarincias individuais e global com parmetro de regularizao \lambda.

python
Copy
class GDARegularized:
    def __init__(self, lambda_=0.5):
        self.lambda_ = lambda_
    
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.priors = {}
        self.covs = {}
        global_cov = np.cov(X.T)
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
            self.covs[c] = (1 - self.lambda_) * np.cov(X_c.T) + self.lambda_ * global_cov
    
    def predict(self, X):
        posteriors = []
        for c in self.classes:
            diff = X - self.means[c]
            inv_cov = np.linalg.inv(self.covs[c])
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            norm = 1 / np.sqrt((2 * np.pi)**2 * np.linalg.det(self.covs[c]))
            posteriors.append(norm * np.exp(exponent) * self.priors[c])
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

class NaiveBayes:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.vars = {}
        self.priors = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.vars[c] = np.var(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
    
    def predict(self, X):
        posteriors = []
        for c in self.classes:
            diff = X - self.means[c]
            exponent = -0.5 * np.sum((diff ** 2) / self.vars[c], axis=1)
            norm = 1 / np.sqrt((2 * np.pi) * np.prod(self.vars[c]))
            posteriors.append(norm * np.exp(exponent) * self.priors[c])
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

Passo 5: Validao por Simulao de Monte Carlo (R = 500 rodadas)

Vamos implementar a validao robusta dos 5 modelos usando Monte Carlo, com R = 500 rodadas de treino/teste (80/20). Para cada rodada, calcularemos a acurcia e armazenaremos os resultados para anlise estatstica.

Implementao em Python

Copy
import numpy as np
from tqdm import tqdm  # Opcional: para barra de progresso

# Dados j carregados e pr-processados (X, y)
R = 500  # Nmero de rodadas
models = {
    "MQO Tradicional": MQOMultinomial(),
    "GDA (Covarincias diferentes)": GDA(),
    "LDA (Covarincias iguais)": LDA(),
    "GDA Matriz Agregada": GDAAggregated(),
    "Naive Bayes": NaiveBayes()
}

# Dicionrio para armazenar acurcias
accuracies = {name: [] for name in models.keys()}

# Simulao de Monte Carlo
for _ in tqdm(range(R)):
    # Particionamento aleatrio (80/20)
    indices = np.random.permutation(len(X))
    train_size = int(0.8 * len(X))
    X_train, y_train = X[indices[:train_size]], y[indices[:train_size]]
    X_test, y_test = X[indices[train_size:]], y[indices[train_size:]]
    
    # One-hot encoding para MQO (apenas nesta rodada)
    Y_train_onehot = np.zeros((len(y_train), 5))
    Y_train_onehot[np.arange(len(y_train)), y_train-1] = 1
    
    # Avaliar cada modelo
    for name, model in models.items():
        if name == "MQO Tradicional":
            model.fit(X_train, Y_train_onehot)
        else:
            model.fit(X_train, y_train)
        y_pred = model.predict(X_test)
        accuracies[name].append(np.mean(y_pred == y_test))

# Estatsticas finais
print("\n=== Acurcia Mdia e Desvio Padro ===")
for name, acc_list in accuracies.items():
    print(f"{name}: {np.mean(acc_list):.2%}  {np.std(acc_list):.2%}")
