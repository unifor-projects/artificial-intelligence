Aqui está uma solução completa usando KNN (K-Nearest Neighbors), que é um algoritmo simples e eficaz para classificação:

import numpy as np
import matplotlib.pyplot as plt

# Carregar os dados
dados = np.loadtxt("TrabalhoAV1/dados/EMGsDataset.csv", delimiter=",")

# Separar características e rótulos
X = dados[:2, :].T  # Transposta para ter formato (50000, 2)
y = dados[2, :].astype(int)  # Converter para inteiro

# 1. Pré-processamento dos dados
# Normalizar os dados (importante para KNN)
X_normalized = (X - X.min(axis=0)) / (X.max(axis=0) - X.min(axis=0))

# 2. Dividir em conjuntos de treino e teste (80% treino, 20% teste)
np.random.seed(42)
indices = np.random.permutation(len(X))
train_size = int(0.8 * len(X))

X_train = X_normalized[indices[:train_size]]
y_train = y[indices[:train_size]]
X_test = X_normalized[indices[train_size:]]
y_test = y[indices[train_size:]]

# 3. Implementação do KNN
def knn_predict(X_train, y_train, X_test, k=5):
    y_pred = np.zeros(X_test.shape[0], dtype=int)
    
    for i, test_sample in enumerate(X_test):
        # Calcular distâncias euclidianas
        distances = np.sqrt(np.sum((X_train - test_sample)**2, axis=1))
        
        # Obter os k vizinhos mais próximos
        k_indices = np.argpartition(distances, k)[:k]
        k_nearest_labels = y_train[k_indices]
        
        # Votação para determinar a classe
        y_pred[i] = np.bincount(k_nearest_labels).argmax()
    
    return y_pred

# 4. Fazer previsões e avaliar o modelo
k = 5  # Número de vizinhos
y_pred = knn_predict(X_train, y_train, X_test, k)

# Calcular acurácia
accuracy = np.mean(y_pred == y_test)
print(f"Acurácia do KNN (k={k}): {accuracy:.2%}")

# 5. Visualização dos resultados
# Plotar os dados de treino
plt.figure(figsize=(12, 6))
plt.subplot(1, 2, 1)
for class_id in np.unique(y_train):
    plt.scatter(X_train[y_train == class_id, 0], 
                X_train[y_train == class_id, 1], 
                label=f'Classe {class_id}', alpha=0.6)
plt.title('Dados de Treino (Normalizados)')
plt.xlabel('Sensor 1 - Corrugador do Supercílio')
plt.ylabel('Sensor 2 - Zigomático Maior')
plt.legend()

# Plotar os dados de teste com as previsões
plt.subplot(1, 2, 2)
for class_id in np.unique(y_test):
    plt.scatter(X_test[y_test == class_id, 0], 
                X_test[y_test == class_id, 1], 
                label=f'Classe {class_id}', alpha=0.6)
plt.title('Dados de Teste com Previsões')
plt.xlabel('Sensor 1 - Corrugador do Supercílio')
plt.legend()

plt.tight_layout()
plt.show()

# Matriz de confusão
def plot_confusion_matrix(y_true, y_pred, classes):
    cm = np.zeros((len(classes), len(classes)), dtype=int)
    
    for i in range(len(y_true)):
        cm[y_true[i]-1, y_pred[i]-1] += 1
    
    plt.figure(figsize=(8, 6))
    plt.imshow(cm, interpolation='nearest', cmap=plt.cm.Blues)
    plt.title('Matriz de Confusão')
    plt.colorbar()
    
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes)
    plt.yticks(tick_marks, classes)
    
    for i in range(len(classes)):
        for j in range(len(classes)):
            plt.text(j, i, str(cm[i, j]),
                     horizontalalignment="center",
                     color="white" if cm[i, j] > cm.max()/2 else "black")
    
    plt.ylabel('Rótulo Verdadeiro')
    plt.xlabel('Rótulo Predito')
    plt.tight_layout()
    plt.show()

plot_confusion_matrix(y_test, y_pred, ['1-Neutro', '2-Sorriso', '3-Sobrancelhas', '4-Surpreso', '5-Rabugento'])

Entendido! Vou organizar os dados conforme solicitado nas duas formas diferentes, dependendo do modelo que será utilizado.

1. Para modelos que usam Mínimos Quadrados Ordinários (MQO):

  • X deve ter dimensão N×pN \times pN×p (50.000 amostras × 2 características)

  • Y deve ter dimensão N×CN \times CN×C (50.000 amostras × 5 classes), em formato one-hot (codificação binária).

2. Para modelos Gaussianos Bayesianos:

  • X deve ter dimensão p×Np \times Np×N (2 características × 50.000 amostras)

  • Y deve ter dimensão C×NC \times NC×N (5 classes × 50.000 amostras), também em formato one-hot.

import numpy as np

# Carregar os dados
dados = np.loadtxt("TrabalhoAV1/dados/EMGsDataset.csv", delimiter=",")

# Extrair características (sensores) e rótulos
sensor1 = dados[0, :]  # Corrugador do Supercílio (1ª linha)
sensor2 = dados[1, :]  # Zigomático Maior (2ª linha)
rotulos = dados[2, :].astype(int)  # Classes (3ª linha)

N = dados.shape[1]  # Número de amostras (50.000)
p = 2  # Número de características (2 sensores)
C = 5  # Número de classes (5 expressões faciais)

# --------------------------------------------
# 1. Formato para MQO (Regressão, SVM, etc.)
# --------------------------------------------
X_mqo = np.column_stack((sensor1, sensor2))  # Formato (N × p)

# Converter Y para one-hot (N × C)
Y_mqo = np.zeros((N, C))
Y_mqo[np.arange(N), rotulos - 1] = 1  # Subtrai 1 porque as classes vão de 1 a 5

# --------------------------------------------
# 2. Formato para Modelos Gaussianos Bayesianos
# --------------------------------------------
X_bayes = np.vstack((sensor1, sensor2))  # Formato (p × N)

# Converter Y para one-hot (C × N)
Y_bayes = np.zeros((C, N))
Y_bayes[rotulos - 1, np.arange(N)] = 1

# --------------------------------------------
# Verificação das dimensões
# --------------------------------------------
print("Formato MQO:")
print("X_mqo shape:", X_mqo.shape)  # Deve ser (50000, 2)
print("Y_mqo shape:", Y_mqo.shape)  # Deve ser (50000, 5)

print("\nFormato Bayesiano:")
print("X_bayes shape:", X_bayes.shape)  # Deve ser (2, 50000)
print("Y_bayes shape:", Y_bayes.shape)  # Deve ser (5, 50000)

    • X_bayes é uma matriz onde cada coluna é uma amostra (formato padrão em estatística bayesiana).

    • Y_bayes também está em one-hot, mas com as classes nas linhas (formato C × N).

Quais suas recomendações quanto aos modelos de gauss bayas

Para o problema de classificação de expressões faciais usando os dados de EMG, modelos Gaussianos Bayesianos podem ser uma boa escolha, especialmente devido à natureza contínua dos sinais dos sensores e à possível distribuição normal dos dados. Aqui estão minhas recomendações:

1. Análise de Adequação dos Modelos Gaussianos Bayesianos

  • Prós:

    • Eficiência computacional: Rápidos para treinar em datasets médios (50k amostras).

    • Interpretabilidade: Probabilidades claras por classe.

    • Bom desempenho se os dados seguirem distribuição normal ou forem transformáveis (ex.: log-transform).

  • Contras:

    • Sensibilidade a outliers: Requer normalização robusta.

    • Pressuposto de independência: Assume que as features (sensores) são independentes, o que pode não ser verdade (músculos faciais podem ativar juntos).

class GaussianNaiveBayes:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.mean = {}
        self.var = {}
        self.prior = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.mean[c] = np.mean(X_c, axis=0)
            self.var[c] = np.var(X_c, axis=0)
            self.prior[c] = X_c.shape[0] / X.shape[0]
    
    def predict(self, X):
        posteriors = []
        
        for c in self.classes:
            prior = np.log(self.prior[c])
            likelihood = -0.5 * np.sum(np.log(2 * np.pi * self.var[c]) + 
                                     ((X - self.mean[c]) ** 2) / self.var[c], axis=1)
            posterior = prior + likelihood
            posteriors.append(posterior)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

B. Análise de Discriminante Gaussiano (GDA)

  • Hipótese: Assume que todas as classes compartilham a mesma matriz de covariância.

  • Vantagem: Captura correlações entre os sensores.

  • Código (extensão do GNB):

class GaussianDiscriminantAnalysis:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.mean = {}
        self.shared_cov = np.cov(X.T)  # Covariância compartilhada
        self.prior = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.mean[c] = np.mean(X_c, axis=0)
            self.prior[c] = X_c.shape[0] / X.shape[0]
    
    def predict(self, X):
        posteriors = []
        inv_cov = np.linalg.inv(self.shared_cov)
        
        for c in self.classes:
            diff = X - self.mean[c]
            likelihood = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            prior = np.log(self.prior[c])
            posteriors.append(prior + likelihood)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

1. MQO Tradicional (Regressão Multinomial)

Objetivo: Minimizar o erro quadrático para classificação multiclasse.

python
Copy
class MQOMultinomial:
    def fit(self, X, Y):
        self.W = np.linalg.inv(X.T @ X) @ X.T @ Y  # Solução fechada: W = (X^T X)^-1 X^T Y
    
    def predict(self, X):
        scores = X @ self.W
        return np.argmax(scores, axis=1) + 1  # +1 para classes 1-5

# Treinar

2. Classificador Gaussiano Tradicional (GDA)

Assume covariâncias diferentes por classe.

python
Copy
class GDA:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.covs = {}
        self.priors = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.covs[c] = np.cov(X_c.T)  # Covariância por classe
            self.priors[c] = len(X_c) / len(X)
    
    def predict(self, X):
        posteriors = []
        for c in self.classes:
            diff = X - self.means[c]
            inv_cov = np.linalg.inv(self.covs[c])
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            norm = 1 / np.sqrt((2 * np.pi)**2 * np.linalg.det(self.covs[c]))
            likelihood = norm * np.exp(exponent)
            posteriors.append(likelihood * self.priors[c])
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

3. Classificador Gaussiano com Covariâncias Iguais (LDA)

Assume mesma covariância para todas as classes.

python
Copy
class LDA:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.priors = {}
        self.shared_cov = np.zeros((X.shape[1], X.shape[1]))
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
            self.shared_cov += (len(X_c) - 1) * np.cov(X_c.T)
        
        self.shared_cov /= (len(X) - len(self.classes))  # Média ponderada
    
    def predict(self, X):
        posteriors = []
        inv_cov = np.linalg.inv(self.shared_cov)
        for c in self.classes:
            diff = X - self.means[c]
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            prior = np.log(self.priors[c])
            posteriors.append(prior + exponent)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

Usa uma única covariância agregada (média das covariâncias por classe).

class GDAAggregated:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.priors = {}
        covs = []
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
            covs.append(np.cov(X_c.T))
        
        self.aggregated_cov = np.mean(covs, axis=0)  # Média simples
    
    def predict(self, X):
        posteriors = []
        inv_cov = np.linalg.inv(self.aggregated_cov)
        for c in self.classes:
            diff = X - self.means[c]
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            prior = np.log(self.priors[c])
            posteriors.append(prior + exponent)
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

5. Classificador Gaussiano Regularizado (Friedman)

Combina covariâncias individuais e global com parâmetro de regularização λ\lambdaλ.

python
Copy
class GDARegularized:
    def __init__(self, lambda_=0.5):
        self.lambda_ = lambda_
    
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.priors = {}
        self.covs = {}
        global_cov = np.cov(X.T)
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
            self.covs[c] = (1 - self.lambda_) * np.cov(X_c.T) + self.lambda_ * global_cov
    
    def predict(self, X):
        posteriors = []
        for c in self.classes:
            diff = X - self.means[c]
            inv_cov = np.linalg.inv(self.covs[c])
            exponent = -0.5 * np.sum(diff @ inv_cov * diff, axis=1)
            norm = 1 / np.sqrt((2 * np.pi)**2 * np.linalg.det(self.covs[c]))
            posteriors.append(norm * np.exp(exponent) * self.priors[c])
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

6. Classificador de Bayes Ingênuo (Naive Bayes)

Assume independência entre features.

python
Copy
class NaiveBayes:
    def fit(self, X, y):
        self.classes = np.unique(y)
        self.means = {}
        self.vars = {}
        self.priors = {}
        
        for c in self.classes:
            X_c = X[y == c]
            self.means[c] = np.mean(X_c, axis=0)
            self.vars[c] = np.var(X_c, axis=0)
            self.priors[c] = len(X_c) / len(X)
    
    def predict(self, X):
        posteriors = []
        for c in self.classes:
            diff = X - self.means[c]
            exponent = -0.5 * np.sum((diff ** 2) / self.vars[c], axis=1)
            norm = 1 / np.sqrt((2 * np.pi) * np.prod(self.vars[c]))
            posteriors.append(norm * np.exp(exponent) * self.priors[c])
        
        return self.classes[np.argmax(np.array(posteriors), axis=0)]

Análise dos Resultados

  1. Interpretação de λ:

    • λ = 0: Usa apenas a covariância individual de cada classe (equivalente ao GDA tradicional).

      • Prós: Melhor ajuste aos dados de treino.

      • Contras: Risco de overfitting se as classes tiverem poucas amostras.

    • λ = 1: Usa apenas a covariância global (equivalente ao LDA).

      • Prós: Reduz variância, ideal para dados com poucas amostras por classe.

      • Contras: Pode subajustar (underfitting) se as classes tiverem covariâncias muito diferentes.

Agora a etapa 5: 5. Para validar os modelos utilizados na tarefa de classificação, sua equipe deve projetar a validação utilizando simulações por Monte Carlo. Nessa etapa, defina a quantidade de rodadas da simulação como R = 500. Em cada rodada, realize o particionamento dos dados, utilizando 80% para treinamento e 20% para teste. A medida de desempenho de cada um dos cinco modelos diferentes deve ser a acurácia (taxa de acerto), e cada valor obtido deve ser armazenado em uma lista.

Passo 5: Validação por Simulação de Monte Carlo (R = 500 rodadas)

Vamos implementar a validação robusta dos 5 modelos usando Monte Carlo, com R = 500 rodadas de treino/teste (80/20). Para cada rodada, calcularemos a acurácia e armazenaremos os resultados para análise estatística.

Implementação em Python

python
Copy
import numpy as np
from tqdm import tqdm  # Opcional: para barra de progresso

# Dados já carregados e pré-processados (X, y)
R = 500  # Número de rodadas
models = {
    "MQO Tradicional": MQOMultinomial(),
    "GDA (Covariâncias diferentes)": GDA(),
    "LDA (Covariâncias iguais)": LDA(),
    "GDA Matriz Agregada": GDAAggregated(),
    "Naive Bayes": NaiveBayes()
}

# Dicionário para armazenar acurácias
accuracies = {name: [] for name in models.keys()}

# Simulação de Monte Carlo
for _ in tqdm(range(R)):
    # Particionamento aleatório (80/20)
    indices = np.random.permutation(len(X))
    train_size = int(0.8 * len(X))
    X_train, y_train = X[indices[:train_size]], y[indices[:train_size]]
    X_test, y_test = X[indices[train_size:]], y[indices[train_size:]]
    
    # One-hot encoding para MQO (apenas nesta rodada)
    Y_train_onehot = np.zeros((len(y_train), 5))
    Y_train_onehot[np.arange(len(y_train)), y_train-1] = 1
    
    # Avaliar cada modelo
    for name, model in models.items():
        if name == "MQO Tradicional":
            model.fit(X_train, Y_train_onehot)
        else:
            model.fit(X_train, y_train)
        y_pred = model.predict(X_test)
        accuracies[name].append(np.mean(y_pred == y_test))

# Estatísticas finais
print("\n=== Acurácia Média e Desvio Padrão ===")
for name, acc_list in accuracies.items():
    print(f"{name}: {np.mean(acc_list):.2%} ± {np.std(acc_list):.2%}")

Por que Monte Carlo?

  • Vantagens:

    • Captura a variabilidade do particionamento dos dados.

    • Mais confiável que uma única divisão treino/teste.

  • Comparação com k-fold CV:

    • Monte Carlo é mais flexível (permute proporções variáveis de treino/teste).

    • k-fold tende a ser menos variável para datasets pequenos.

Recomendações Finais

  • Modelo Vencedor: Use o GDA (Covariâncias diferentes) se a diferença para os outros for estatisticamente significativa (ex.: teste t).

  • Regularização: Adicione Friedman (λ = 0.5) ao GDA para reduzir overfitting.

  • Próximos Passos:

    • Testar modelos não-lineares (ex.: SVM com kernel RBF) se a acurácia for insuficiente.

    • Extrair features temporais (média, variância) para melhorar separabilidade.
